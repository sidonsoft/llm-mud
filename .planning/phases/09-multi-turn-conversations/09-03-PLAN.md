def add_message(self, content: str, activity_type=ActivityType.IDLE, timestamp=None) -> None:
    """Add a message to short-term memory with relevance scoring."""
    entry = MemoryEntry(content=content, timestamp=timestamp, activity_type=activity_type)
    entry.relevance_score = self._calculate_relevance(entry)
    self.short_term_memory.append(entry)
